# Plan: Add Biji Relai (Palm Seed Tonnage) Data Input

## Context
Biji Relai (palm seeds) are collected alongside palm fruit. Unlike daily entries (per team leader), Biji Relai is tracked **per block per day** — there is no team leader association. The monthly Excel export already has Biji Relai rows in the template but they are currently blank (hardcoded to 0). This plan adds the data input UI and wires it into the export.

---

## Step 1: Database — Create `biji_relai` table

**File:** `scripts/biji_relai_migration.sql` (new)

SQL to run in Supabase SQL Editor:
```sql
CREATE TABLE IF NOT EXISTS biji_relai (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE NOT NULL,
  plantation_id UUID REFERENCES plantations(id) ON DELETE CASCADE NOT NULL,
  date DATE NOT NULL,
  tons NUMERIC,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(user_id, plantation_id, date)
);

ALTER TABLE biji_relai ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Users can manage their own biji_relai"
  ON biji_relai FOR ALL
  USING (auth.uid() = user_id);

CREATE INDEX idx_biji_relai_user_block_date
  ON biji_relai(user_id, plantation_id, date);
```

Key design decisions:
- **Unique constraint on (user_id, plantation_id, date)** — one seed tonnage entry per block per day
- **No team_leader_id** — Biji Relai is block-level, not team-leader-level
- **RLS enabled** — same pattern as other tables

### New data loading function:
```ts
async function loadBijiRelai(blockId: string) {
  const { data } = await supabase
    .from("biji_relai")
    .select("*")
    .eq("user_id", user!.id)
    .eq("plantation_id", blockId)
    .order("date", { ascending: false });
  setBijiRelaiEntries((data || []) as BijiRelai[]);
}
```

Call `loadBijiRelai(selectedBlock)` when `selectedBlock` changes (in the existing `useEffect` that calls `loadOrgChartData`).

### New submit handler:
```ts
async function handleSubmitBijiRelai(e: React.FormEvent) {
  e.preventDefault();
  if (!user || !selectedBlock) return;
  setBijiRelaiSaving(true);

  const entryData = {
    user_id: user.id,
    plantation_id: selectedBlock,
    date: bijiRelaiDate,
    tons: bijiRelaiTons ? parseFloat(bijiRelaiTons) : null,
  };

  // Upsert: check existing entry for same date + block
  const existing = bijiRelaiEntries.find(
    (e) => e.date === bijiRelaiDate && e.plantation_id === selectedBlock
  );

  if (existing) {
    await supabase.from("biji_relai").update(entryData).eq("id", existing.id);
  } else {
    await supabase.from("biji_relai").insert(entryData);
  }

  setBijiRelaiSaving(false);
  setBijiRelaiSavedId("saved");
  loadBijiRelai(selectedBlock);
}
```

```ts
// Build bijiRelaiByDay map
const bijiByDay: Record<number, number> = {};
let bijiCum = 0;
for (const br of bijiRelaiData || []) {
  const day = dayOf(br.date);
  if (br.tons !== null && br.tons !== undefined) {
    bijiByDay[day] = round2(Number(br.tons));
  }
}

// Populate Biji Relai daily values
for (let d = 1; d <= days; d++) {
  const dc = dayCol(d);
  if (bijiByDay[d] !== undefined) {
    rows[R_BIJI][dc] = bijiByDay[d];        // Hari ini
    bijiCum = round2(bijiCum + bijiByDay[d]);
  }
  if (bijiCum > 0) {
    rows[R_BIJI_CUM][dc] = bijiCum;          // Hingga Hari Ini
  }
}

// Update totals
let totalBiji = 0;
for (let d = 1; d <= days; d++) {
  if (bijiByDay[d] !== undefined) totalBiji = round2(totalBiji + bijiByDay[d]);
}
rows[R_BIJI][38] = totalBiji;     // AM column — Hari ini total
rows[R_BIJI_CUM][38] = bijiCum;   // AM column — cumulative total
```
